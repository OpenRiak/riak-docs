Francisco

ps: my current workaround: copy/paste the Module into user.rb - this works

module Boolean
 def self.ripple\_cast(value)
 case value
 when NilClass
 nil
 when Numeric
 !value.zero?
 when TrueClass, FalseClass
 value
 when /^\s\*t/i
 true
 when /^\s\*f/i
 false
 else
 value.present?
 end
 end
end
