> %% Set some vars
> {Bucket, Key, Index} = {<<"test\_users">>, <<"aldarkusa">>, 
> [{"qayerda\_bin", "Qozoqta"}]}.
> 
> %% Connect to riak
> {ok, Pid} = riakc\_pb\_socket:start\_link("127.0.0.1", 8087).
> 
> %% Create a new riak object
> Obj = riakc\_obj:new(Bucket, Key, <<"Yomonlarni aldap yurgan Aldarkusa">>).
> 
> %% Create a new metadata with secondary index
> Meta = dict:store(<<"index">>, Index, dict:new()).
> 
> %% Update the object we've created before
> NewObj = riakc\_obj:update\_metadata(Obj, Meta).
> 
> %% Finally, put/insert the updated new object
> riakc\_pb\_socket:put(Pid, NewObj).
>
