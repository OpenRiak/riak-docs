&gt;&gt; ps: my current workaround: copy/paste the Module into user.rb - this works
&gt;&gt;
&gt;&gt; module Boolean
&gt;&gt;  def self.ripple\\_cast(value)
&gt;&gt;    case value
&gt;&gt;    when NilClass
&gt;&gt;      nil
&gt;&gt;    when Numeric
&gt;&gt;      !value.zero?
&gt;&gt;    when TrueClass, FalseClass
&gt;&gt;      value
&gt;&gt;    when /^\\s\\*t/i
&gt;&gt;      true
&gt;&gt;    when /^\\s\\*f/i
&gt;&gt;      false
&gt;&gt;    else
&gt;&gt;      value.present?
&gt;&gt;    end
&gt;&gt;  end
&gt;&gt; end
