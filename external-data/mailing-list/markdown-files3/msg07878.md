Input: [{Country1:[obj1, obj2..], Country2:[[obj3, obj4]},
{Country1:[[obj5, obj6], Country3:[[obj7, obj8]}
Output: [{Country1:[obj1, obj2,obj5, obj6], Country2:[[obj3,
obj4], Country3:[[obj7, obj8]}]
