Jason Strutz Mon, 18 Nov 2013 12:43:15 -0800

             "timeout": 86400000,

                 "function":"map_identity"}
               },
               {"reduce":{
                  "language":"erlang",
                   "module":"riak_kv_mapreduce",
                   "function":"reduce_count_inputs"}
               }
]}' -H "Content-Type: application/json" http://riak01:8098/mapred
