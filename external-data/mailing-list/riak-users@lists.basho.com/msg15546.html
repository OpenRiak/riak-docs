    {
        var map = buck.MapReduce(new ErlangMapPhase("unity", "map_phase"), new

        Debug.Log("Start");
        yield return map.WaitUntilDone();
        if(map.isSuccessful)
        foreach (var item in map.GetResult<Interactor>())
        {
            Debug.Log("Ok");
            foreach (var pair in item.Fields) Debug.Log(pair.Key + " = " + 
pair.Value.ToString());
        }
        else
        {
            Debug.Log("Fail");
        }
    }
