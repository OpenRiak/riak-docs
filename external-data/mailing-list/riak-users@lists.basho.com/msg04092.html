>>>>> 
>>>>> db.add('users').map(function(v) {
>>>>> v =

180 || v.exempt) && v.acl && v.acl.state === '1101'
>>>>> && !v.expired)
{
>>>>> if (v.address) {
>>>>> result.push(v.address.street);
>>>>>
}
>>>>> }
>>>>> return result;
>>>>> }).run()
>>>>> 
>>>>> riak-ql is
basically adding some query sugar (where, &&) on top of
>>>>>
