in one query. Here it is:

map\_kv\_pairs ( Obj, \_, \_ ) ->
 case riak\_object:is\_robject(Obj) of
 true -> [ {riak\_object:key(Obj), riak\_object:get\_value(Obj)} ] ;
 false -> []
 end.
