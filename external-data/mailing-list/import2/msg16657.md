 192 --key ${RIAK\_ADMIN\_SECRET} \ 
 193 --acl private \ 
 194 -- -s -v -x social.dev.pryvy.com:50201 \
 195 -X PUT http://social-media.cs.pryvy.com/ > /dev/null 2>&1
 196 echo " created social-media bucket" 
 197 
 198 eval "bin/s3curl.pl --debug --id ${RIAK\_ADMIN\_KEY} --key 
${RIAK\_ADMIN\_SECRET} -- -s -v -x social.dev.pryvy.com:50201 -H 
'x-amz-grant-full-control: id=\"${RIAK\_A DMIN\_ID}\"' -H 'x-amz-grant-write: 
id=\"${RIAK\_SOCIAL\_ID}\"' -X PUT http://social-media.cs.pryvy.com/?acl" > 
/dev/null 2>&1 
 199 echo " applied social bucket policy"
