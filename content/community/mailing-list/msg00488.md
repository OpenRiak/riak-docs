&gt; ps: my current workaround: copy/paste the Module into user.rb - this works

&gt; module Boolean
&gt; def self.ripple\\_cast(value)
&gt; case value
&gt; when NilClass
&gt; nil
&gt; when Numeric
&gt; !value.zero?
&gt; when TrueClass, FalseClass
&gt; value
&gt; when /^\\s\\*t/i
&gt; true
&gt; when /^\\s\\*f/i
&gt; false
&gt; else
&gt; value.present?
&gt; end
&gt; end
&gt; end
