 z = create :zid
 i = create :identity, zid\\_id: z.key
 Identity.find\\_by\\_index('zid\\_id', z.key).count.should eq(1)

 i.email = 'uni...@unique.com'
 i.save
 Identity.find\\_by\\_index('zid\\_id', z.key).count.should eq(1)
 end
 end
